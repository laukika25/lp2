        Decimal ans = (Celsius * 9/5) + 32;

    }

        Decimal ans = (Fahrenheit - 32) * 5/9;
        System.debug('Fahrenheit: ' + Fahrenheit + '째F = Celsius: ' + ans + '째C');
    }
}
